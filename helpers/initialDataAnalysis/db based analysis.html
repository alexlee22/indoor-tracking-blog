#This gives us an idea of the time domain that the data we have is interested in.
def timeStats(df, showprint=True):
    minT = df.time.min()
    maxT = df.time.max()
    rangeT = maxT-minT
    if showprint:
        print minT,   "First Recording in this set"
        print maxT,   "Last Recording in this set"
        print rangeT, "Range covered"
    return (minT,maxT,rangeT)

def timeSliced(startTime, windowLength, df):
    if type(startTime) is str:
        startTime = dateutil.parser.parse(startTime)
    delta = datetime.timedelta(seconds=windowLength)
    endTime = startTime+delta
    df = df[(df.time>startTime) & 
            (df.time<endTime)]
    
    centroidweighted = {"x":df.xWeighted.mean(), "y":df.yWeighted.mean()}
    centroid         = {"x":df.x.mean(), "y":df.y.mean()}
    
    title = "Detections for person 297 from {} to {} in steps of {} seconds".format(startTime, startTime+delta, windowLength)
    plt.title(title)
    img = imread("map.png")
    
    plt.scatter(df.x, df.y, zorder=1, s=abs(df.rssiAdj)*100, alpha=0.2)
    
    plt.scatter(centroid["x"],         centroid["y"],         c="g", zorder=2, s=100)
    plt.scatter(centroidweighted["x"], centroidweighted["y"], c="r", zorder=2, s=100)
    plt.plot([centroid["x"], centroidweighted["x"]], [centroid["y"], centroidweighted["y"]], 'k-', lw=2)
    
    plt.imshow(img, zorder=0, extent=[-1300, 72000, -24000, 41000])
    #                                [left,  right, bottom, top  ]
    # plt.show() #don't show, we need to save
    plt.savefig("frames/dbAnim/foo"+str(startTime)+".png", bbox_inches='tight')
    plt.clf()
