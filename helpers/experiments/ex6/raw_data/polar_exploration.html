def convert_timestring(row):
    return dateutil.parser.parse(row.time)

N = 1000 # number of data points

# data = 3.0*np.sin(t+0.001) + 0.5 + np.random.randn(N) # create artificial data with noise

rssiShift = -2.5
pitchSpread = 0.9
guess_phase = np.deg2rad(300) # just twiddle these numbers until it looks right
guess_mean = np.mean(hockey.rssi)
guess_std = 3*np.std(hockey.rssi)/(2**0.5)

print guess_mean, guess_std, guess_phase

# we'll use this to plot our first estimate. This might already be good enough for you
data_first_guess = guess_std*np.sin((t+guess_phase)* pitchSpread) + guess_mean + rssiShift

plt.plot(estimote.rssi, "." , label='RSSI, estimote')
plt.plot(hockey.rssi, "." , label='RSSI, hockey')
plt.plot(data_first_guess, label='first guess')
